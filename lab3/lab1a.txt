p2: check if a number is prime
var x;
var isPrime = true;
Console.ReadInteger(x);
if (x <= 1) {
    isPrime = false;
}

if (isPrime == true) {
    for (var d = 2; d * d <= x; d++) {
        if (x % d == 0) {
            isPrime = false;
            break;
        }
    }
}

if (isPrime == true) {
    Console.Print("The number is prime);
}
else {
    Console.Print("The number is NOT prime);
}

p3: the sum of n numbers

var n, x;
var arr = new Array<int>();
var sum = 0;
Console.ReadInteger(n);
for(var i = 1; i <= n; i++) {
    Console.ReadInteger(x);
    arr[i] = x;
    sum = sum + arr[i];
}

Console.Print(sum);
